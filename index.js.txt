import { useState } from 'react';
import Head from 'next/head';
export default function Home() {
  const [artist, setArtist] = useState('');
  const [theme, setTheme] = useState('');
  const [startBpm, setStartBpm] = useState(90);
  const [playlistLength, setPlaylistLength] = useState(8);
  const [playlist, setPlaylist] = useState([]);
  const [loading, setLoading] = useState(false);
  const generatePlaylist = async () => {
    setLoading(true);
    try {
      const response = await fetch('/api/generate-playlist', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ artist, theme, startBpm, playlistLength }),
      });
      const data = await response.json();
      setPlaylist(data.songs || []);
    } catch (error) {
      console.error('Error:', error);
    }
    setLoading(false);
  };
  return (
    
      <Head>
        <title>Spin Playlist Generator</title>
        <meta name=&quot;description&quot; content=&quot;Generate fire playlists for your spin class&quot; />
      </Head>
      <div className=&quot;min-h-screen bg-gradient-to-br from-purple-900 via-black to-pink-900 text-white p-8&quot;>
        <div className=&quot;max-w-2xl mx-auto&quot;>
          <h1 className=&quot;text-5xl font-bold mb-2 text-transparent bg-clip-text bg-gradient-to-r from-pink-400 to-purple-400&quot;>
            🔥 Spin Playlist Generator
          </h1>
          <p className=&quot;text-gray-300 mb-8&quot;>Create fire playlists with perfect BPM progression</p>
          <div className=&quot;bg-gray-900 bg-opacity-50 rounded-lg p-8 backdrop-blur-md border-500 border-opacity-30&quot;>
            <input
              type="text"
              placeholder="Artist name (e.g., Drake, Pitbull)"
              value={artist}
              onChange={(e) => setArtist(e.target.value)}
              className="w-full mb-4 p-3 bg-gray-800 border border-pink-500 rounded text-white placeholder-gray-400 focus:outline-none focus:border-pink-400"
            />
            <input
              type="text"
              placeholder="Theme name (e.g., Drake Takeover)"
              value={theme}
              onChange={(e) => setTheme(e.target.value)}
              className="w-full mb-4 p-3 bg-gray-800 border border-pink-500 rounded text-white placeholder-gray-400 focus:outline-none focus:border-pink-400"
            />
            <input
              type="number"
              placeholder="Starting BPM"
              value={startBpm}
              onChange={(e) => setStartBpm(Number(e.target.value))}
              className="w-full mb-4 p-3 bg-gray-800 border border-pink-500 rounded text-white placeholder-gray-400 focus:outline-none focus:border-pink-400"
            />
            <input
              type="number"
              placeholder="Playlist length (songs)"
              value={playlistLength}
              onChange={(e) => setPlaylistLength(Number(e.target.value))}
              className="w-full mb-6 p-3 bg-gray-800 border border-pink-500 rounded text-white placeholder-gray-400 focus:outline-none focus:border-pink-400"
            />
            <button
              onClick={generatePlaylist}
              disabled={loading}
              className="w-full bg-gradient-to-r from-pink-500 to-purple-500 hover:from-pink-600 hover:to-purple-600 disabled:opacity-50 font-bold py-3 px-4 rounded transition"
            >
              {loading ? 'Generating...' : '🎵 Generate Playlist'}
            </button>
          </div>
          {playlist.length > 0 && (
            <div className=&quot;mt-8&quot;>
              <h2 className=&quot;text-2xl font-bold mb-4&quot;>📋 Your Playlist: {theme}</h2>
              <div className=&quot;space-y-3&quot;>
                {playlist.map((song, i) => (
                  <a
                    key={i}
                    href={`https://soundcloud.com/search?q=${encodeURIComponent(song.name)}+${encodeURIComponent(song.artist)}`}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="block p-4 bg-gray-900 border-l-4 border-pink-500 hover:bg-gray-800 transition"
                  >
                    <div className=&quot;flex justify-between items-start&quot;>
                      <div>
                        <p className=&quot;font-bold&quot;>{i + 1}. {song.name}</p>
                        <p className=&quot;text-sm text-gray-400&quot;>{song.artist}</p>
                      </div>
                      <div className=&quot;text-right&quot;>
                        <p className=&quot;font-bold text-pink-400&quot;>{song.bpm} BPM</p>
                        <p className=&quot;text-xs text-gray-400&quot;>{song.energy}</p>
                      </div>
                    </div>
                  </a>
                ))}
              </div>
            </div>
          )}
        </div>
      </div>
    </>
  );
}
